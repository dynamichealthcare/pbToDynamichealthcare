AddProduct
    [Arguments]    ${pstring}
    Log    ${pstring}
    ${pstringArr}=    Split String    ${pstring}    ::
    ${pname}=    Set Variable    ${pstringArr[0]}
    ${pprice}=    Set variable    ${pstringArr[1]}
    ${pcontent}=    Set variable    ${pstringArr[2]}
    ${brandNamex}=    Set variable    ${pstringArr[5]}
    ${pimgs}=    Set variable    ${pstringArr[4]}
    ${pvariants}=    Set Variable    ${pstringArr[3]}
    Log    ${pstringArr[3]}
    ${pvariants}=    Split String    ${pvariants}    |
    Log    ${pvariants}
    ${variantStringLen}=    Get Length    ${pvariants}
    Run Keyword If    ${variantStringLen}>4    AddVarient    @{pvariants}
    ...    ELSE    AddProductForVariant    ${pstring}    X
    :FOR    ${i}    IN RANGE    0    ${variantStringLen}
    \    ${xxx}=    Set Variable    ${pvariants[${i}]}
    \    ${xxx}=    Split String    ${xxx}    !!
    \    Run Keyword If    ${variantStringLen}>0    AddProductForVariant    ${pname} - ${xxx[3]}    ${xxx[2]}    ${xxx[1]}
    \    ...    ${brandNamex}    ${pimgs}
    Run Keyword If    ${variantStringLen}==0    AddProductForVariant    ${pname}    ${pprice}    ${pcontent}    ${brandNamex}
    ...    ${pimgs}
